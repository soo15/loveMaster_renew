Pagination.prototype.getPageListFormat = function () {
  var result = "";

  for (var num = 1; num <= this.total_page; num++) {
    var className = num == this.current_page ? "active" : "";
    result += '<button class="' + className + '">' + num + "</button>";
  }

  return result;
};
